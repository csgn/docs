1    package com.csgn.myapplication
2    
3    import android.graphics.Color
4    import androidx.appcompat.app.AppCompatActivity
5    import android.os.Bundle
6    import android.view.Gravity
7    import android.view.View
8    import android.widget.*
9    import androidx.appcompat.app.AlertDialog
10   import com.google.android.material.floatingactionbutton.FloatingActionButton
11   import com.google.android.material.snackbar.Snackbar
12   
13   class MainActivity : AppCompatActivity() {
14       override fun onCreate(savedInstanceState: Bundle?) {
15           super.onCreate(savedInstanceState)
16           setContentView(R.layout.activity_main)
17   
18   
19           /* TOAST MESSAGE */
20           val normalButton = findViewById<Button>(R.id.normalButton);
21           val specialButton = findViewById<Button>(R.id.specialButton)
22   
23           normalButton.setOnClickListener {
24               Toast.makeText(applicationContext, "Normal Button was clicked!", Toast.LENGTH_SHORT).show()
25           }
26   
27           specialButton.setOnClickListener {
28               val design = layoutInflater.inflate(R.layout.toast_design, null)
29               val toast = Toast(applicationContext)
30               toast.view = design
31               toast.setGravity(Gravity.CENTER_HORIZONTAL or Gravity.CENTER_VERTICAL, 0, 0)
32               toast.duration = Toast.LENGTH_SHORT
33               toast.show()
34           }
35   
36           /* ALERT MESSAGE */
37           val alertButton = findViewById<Button>(R.id.alertButton)
38           alertButton.setOnClickListener {
39               val design = layoutInflater.inflate(R.layout.toast_design, null)
40               val editAlert = design.findViewById<EditText>(R.id.editInput)
41               val alert = AlertDialog.Builder(this)
42   
43               alert.setMessage("Please write a name")
44               alert.setTitle("Alert Dialog")
45               alert.setIcon(android.R.drawable.btn_star)
46               alert.setView(design)
47   
48               alert.setPositiveButton("Save") {dialogRef, i ->
49                   val ctx = editAlert.text.toString()
50                   Toast.makeText(applicationContext, "$ctx is saved.", Toast.LENGTH_SHORT).show()
51               }
52   
53               alert.setNegativeButton("Cancel") {dialogRef, i ->
54                   Toast.makeText(applicationContext, "The text is deleted.", Toast.LENGTH_SHORT).show()
55               }
56               alert.show()
57           }
58   
59           /* SNACKBAR */
60           val snackButton = findViewById<Button>(R.id.snackButton)
61           snackButton.setOnClickListener { i ->
62               val snackBar = Snackbar.make(i, "The message is deleted", Snackbar.LENGTH_SHORT)
63   
64               snackBar.setAction("Continue") {}
65               snackBar.setActionTextColor(Color.RED).setTextColor(Color.BLUE)
66               snackBar.show()
67           }
68   
69           /* POPUPs */
70           val popupButton = findViewById<Button>(R.id.popupButton)
71           popupButton.setOnClickListener {
72               val popupMenu = PopupMenu(this, popupButton)
73               popupMenu.menuInflater.inflate(R.menu.popup_menu, popupMenu.menu)
74               popupMenu.setOnMenuItemClickListener { i->
75                   when (i.itemId) {
76                       R.id.create -> {
77                           Toast.makeText(applicationContext, "Created", Toast.LENGTH_SHORT).show()
78                           true
79                       }
80   
81                       R.id.delete -> {
82                           Toast.makeText(applicationContext, "Deleted", Toast.LENGTH_SHORT).show()
83                           true
84                       }
85   
86                       R.id.cancel -> {
87                           Toast.makeText(applicationContext, "Cancel", Toast.LENGTH_SHORT).show()
88                           true
89                       }
90   
91                       else -> false
92                   }
93               }
94   
95               popupMenu.show()
96           }
97   
98           /* RADIO BOX */
99           val womanRadioButton = findViewById<RadioButton>(R.id.womanRadioButton)
100          womanRadioButton.setOnCheckedChangeListener { buttonView, isChecked ->
101              if (isChecked) {
102                  Toast.makeText(applicationContext, "Woman", Toast.LENGTH_SHORT).show()
103              }
104          }
105  
106          val manRadioButton = findViewById<RadioButton>(R.id.manRadioButton)
107          manRadioButton.setOnCheckedChangeListener { buttonView, isChecked ->
108              if (isChecked) {
109                  Toast.makeText(applicationContext, "Man", Toast.LENGTH_SHORT).show()
110              }
111          }
112  
113          /* CHECK BOX */
114          val checkBox1 = findViewById<CheckBox>(R.id.checkBox1)
115          val checkBox2 = findViewById<CheckBox>(R.id.checkBox2)
116  
117          checkBox1.setOnCheckedChangeListener { checkBoxView, isChecked ->
118              if (isChecked) {
119                  Toast.makeText(applicationContext, "CheckBox 1 is checked", Toast.LENGTH_SHORT).show()
120              } else {
121                  Toast.makeText(applicationContext, "CheckBox 1 is un checked", Toast.LENGTH_SHORT).show()
122              }
123          }
124  
125          checkBox2.setOnCheckedChangeListener { checkBoxView, isChecked ->
126              if (isChecked) {
127                  Toast.makeText(applicationContext, "CheckBox 2 is checked", Toast.LENGTH_SHORT).show()
128              } else {
129                  Toast.makeText(applicationContext, "CheckBox 2 is un checked", Toast.LENGTH_SHORT).show()
130              }
131          }
132  
133          /* PROGRESS BAR */
134          val startProgressBar = findViewById<Button>(R.id.startProgressBar)
135          val stopProgressBar = findViewById<Button>(R.id.stopProgressBar)
136          val progressBar = findViewById<ProgressBar>(R.id.progressBar)
137  
138          startProgressBar.setOnClickListener {
139              progressBar.visibility = View.VISIBLE
140          }
141  
142          stopProgressBar.setOnClickListener {
143              progressBar.visibility = View.INVISIBLE
144          }
145  
146          /* FLOATING ACTION BUTTON */
147          val floatingActionButton = findViewById<FloatingActionButton>(R.id.floatingActionButton)
148          floatingActionButton.setOnClickListener {
149              Toast.makeText(applicationContext, "Floating Button is clicked", Toast.LENGTH_SHORT).show()
150          }
151      }
152  }
